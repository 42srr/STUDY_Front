## 35.2 배열 리터럴 내부에서 사용하는 경우

### 35.2.1 concat

const arr = [...[1, 2], ...[3, 4]];
console.log(arr); // [1, 2, 3, 4]

### 35.2.2 splice

const arr1 = [1, 4];
const arr2 = [2, 3];

arr1.aplice(1, 0, ...arr2);
console.log(arr1); // [1, 2, 3, 4]
